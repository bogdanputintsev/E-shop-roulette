10   /**
11    * @method ProductsKeys|null find($id, $lockMode = null, $lockVersion = null)
12    * @method ProductsKeys|null findOneBy(array $criteria, array $orderBy = null)
13    * @method ProductsKeys[]    findAll()
14    * @method ProductsKeys[]    findBy(array $criteria, array $orderBy = null, $limit = null, $offset = null)
15    */
16   class ProductsKeysRepository extends ServiceEntityRepository implements IProductsKeysRepository
17   {
18       public function __construct(ManagerRegistry $registry)
19       {
20           parent::__construct($registry, ProductsKeys::class);
21       }
22   
23       /**
24        * Use - nastaví stav klíču jako použity podle id
25        */
26       public function Use(?int $id)
27       {
28           $qb = $this->createQueryBuilder('p');
29           $qb->update()
30               ->set('p.is_used', true)
31               ->where('p.id = ' . $id)
32               ->getQuery()->execute();
33       }
34   
35       /**
36        * Use - nastaví stav klíču jako použity podle id
37        */
38       public function isAvailable($id)
39       {
40           if ( $id == "" ){
41               return false;
42           }
43           $qb = $this->createQueryBuilder('p');
44           return $qb  ->select('count(p.id)')
45                       ->where('p.product_id = ' . (int)$id )
46                       ->getQuery()->getSingleResult() > 0;
47       }
48   
49       /**
50        * Vrátí klíče konkrétního produktu, které nebyly použité
51        */
52       public function findUnused($game_id)
53       {
54           $qb = $this->createQueryBuilder('p');
55           $qb
56               -> select('p.id', 'p.value')
57               -> where('p.daily_gift = 0')
58               -> andWhere('p.product_id = '.(int)$game_id)
59               -> andWhere('p.is_used = 0');
60           return $qb->getQuery()->getArrayResult();
61       }
62   
63       /**
64        * Usekey - nastaví stav klíču jako použity podle retezce klice
65        */
66       public function UseKey($msg)
67       {
68           $qb = $this->createQueryBuilder('p');
69           $qb
70               ->update()
71               ->set('p.is_used', 1)
72               ->where('p.value = \'' . $msg . '\'')
73               ->getQuery()->execute();
74       }

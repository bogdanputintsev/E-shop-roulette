3    namespace App\Repository;
4    
5    use App\Entity\Products;
6    use App\Interfaces\IProductsRepository;
7    use Doctrine\Bundle\DoctrineBundle\Repository\ServiceEntityRepository;
8    use Doctrine\Common\Persistence\ManagerRegistry;
9    
10   /**
11    * @method Products|null find($id, $lockMode = null, $lockVersion = null)
12    * @method Products|null findOneBy(array $criteria, array $orderBy = null)
13    * @method Products[]    findAll()
14    * @method Products[]    findBy(array $criteria, array $orderBy = null, $limit = null, $offset = null)
15    */
16   class ProductsRepository extends ServiceEntityRepository implements IProductsRepository
17   {
18       public function __construct(ManagerRegistry $registry)
19       {
20           parent::__construct($registry, Products::class);
21       }
22   
23       /**
24        * findProducts - vrátí informace o produktů
25        */
26       public function findProducts($products)
27       {
28           $qb = $this->createQueryBuilder('p');
29           $qb
30               -> select('p.id', 'p.name', 'p.image_key', 'p.price_sell', 'p.price', 'p.type', 'p.system', 'p.drop_chance')
31               -> where ('p.id IN (:values)')
32               -> setParameter( 'values', explode(',', $products));
33           return $qb->getQuery()->getArrayResult();
34       }
35   
36       /**
37        * Vrátí všechny produkty, které jsou na prodeji
38        */
39       public function findAllProductsInShop()
40       {
41           $qb = $this->createQueryBuilder('p');
42           $qb
43               ->select('p.id', 'p.name', 'p.image_key', 'p.price', 'p.type', 'p.visible')
44               ->where('p.type = 1')
45               ->andWhere('p.visible = 1');
46           return $qb->getQuery()->getArrayResult();
47       }
48   
49       /**
50        * Vrátí všechny účty, které jsou na prodeji
51        */
52       public function findAllAccountsInShop()
53       {
54           $qb = $this->createQueryBuilder('p');
55           $qb
56               ->select('p.id', 'p.name', 'p.image_key', 'p.price', 'p.type', 'p.visible')
57               ->where('p.type = 2')
58               ->andWhere('p.visible = 1');
59           return $qb->getQuery()->getArrayResult();
60       }
61   
62       /**
63        * Vrátí konkrétní produkt
64        */
65       public function findGame($id)
66       {
67           $qb = $this->createQueryBuilder('p');
68           $qb
69               ->where('p.id = '.(int)$id);
70           return $qb->getQuery()->getArrayResult();
71       }

5    use App\Entity\Cases;

9    
10   /**
11    * @method Cases|null find($id, $lockMode = null, $lockVersion = null)
12    * @method Cases|null findOneBy(array $criteria, array $orderBy = null)
13    * @method Cases[]    findAll()
14    * @method Cases[]    findBy(array $criteria, array $orderBy = null, $limit = null, $offset = null)
15    */
16   class CasesRepository extends ServiceEntityRepository implements ICasesRepository
17   {
18       public function __construct(ManagerRegistry $registry)
19       {
20           parent::__construct($registry, Cases::class);
21       }
22   
23       /**
24        * Najde potřebnou informace o konkrétní ruletě
25        */
26       public function build ($id)
27       {
28           $qb = $this->createQueryBuilder('n');
29   
30           $qb ->select('n.id', 'n.name', 'n.description', 'n.price', 'n.products')
31               ->where('n.id = ' . $id);
32           return $qb
33               ->getQuery()
34               ->getArrayResult();
35       }
36   
37       /**
38        *  Nalezne všechny rulety, které jsou propojeny s nějakou hrou
39        */
40       public function findCases()
41       {
42           $qb = $this->createQueryBuilder('c');
43           $qb
44               -> select('c.id', 'c.name', 'c.price', 'c.game_id', 'p.image_key')
45               -> innerJoin('App\Entity\Products', 'p', 'WITH', 'p = c.game_id')
46               -> where('c.game_id IS NOT NULL')
47               -> orderBy( 'c.popularity', 'DESC');
48           return $qb-> getQuery()->getResult();
49       }

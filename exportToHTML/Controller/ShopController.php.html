3    namespace App\Controller;
4    
5    
6    use App\Interfaces\IProductsKeysRepository;
7    use App\Interfaces\IProductsRepository;
8    use App\Interfaces\IUsersRepository;
9    use Symfony\Bundle\FrameworkBundle\Controller\AbstractController;
10   use Symfony\Component\HttpFoundation\Request;
11   use Symfony\Component\HttpFoundation\Response;
12   use Symfony\Component\Routing\Annotation\Route;
13   
14   class ShopController extends AbstractController
15   {
16   
17       /**
18        * Index - zobrazí stránku, která obsahuje všechny produkty, které jsou na prodeji
19        * @Route("/shop", name="shop")
20        * @param IProductsRepository $productsRepository
21        * @return Response
22        */
23       public function index(IProductsRepository $productsRepository)
24       {
25           $products = $productsRepository->findAllProductsInShop();
26           $accounts = $productsRepository->findAllAccountsInShop();
27   
28           return $this->render('shop/index.html.twig', [
29               'controller_name' => 'ShopController',
30               'page' => 'shop',
31               'avatar' =>   $this->get('session')->get('avatar'),
32               'uid' => 'none' ? $this->get('session')->get('id') == "" : $this->get('session')->get('id'),
33               'username' =>  $this->get('session')->get('username'),
34               'balance' => $this->get('session')->get('balance'),
35               'products' => $products,
36               'accounts' => $accounts
37           ]);
38       }
39   
40       /**
41        * Purchase - stránka s konkrétním produktem, kde uživatel si muže koupit k této hře
42        * @Route("/purchase/{id}", name="purchase")
43        * @param Request $request
44        * @param IUsersRepository $usersRepository
45        * @param IProductsRepository $productsRepository
46        * @param IProductsKeysRepository $keysRepository
47        * @return Response
48        */
49       public function purchase(Request $request, IUsersRepository $usersRepository, IProductsRepository $productsRepository, IProductsKeysRepository $keysRepository)
50       {
51           $keysAvailable = $keysRepository->isAvailable($request->get("id"));
52           $game = $productsRepository->findGame($request->get('id'));
53           if($game == null)
54           {
55               return $this->redirect('404.html.twig');
56           }
57   
58           return $this->render('shop/purchase.html.twig', [
59               'controller_name' => 'ShopController',
60               'page' => 'purchase',
61               'avatar' =>   $this->get('session')->get('avatar'),
62               'uid' => 'none' ? $this->get('session')->get('id') == "" : $this->get('session')->get('id'),
63               'username' =>  $this->get('session')->get('username'),
64               'balance' => $this->get('session')->get('id') != "" ? $usersRepository->find($this->get('session')->get('id'))->getBalance() : 0,
65               'keys_available' => $keysAvailable,
66               'game' => $game
67           ]);
68       }
69   }

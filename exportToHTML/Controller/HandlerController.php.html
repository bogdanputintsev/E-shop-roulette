31       {
32           $order = $shopService->buildOrder($_POST['game_id'], $_POST['pay_id'], $productsRepository, $keysRepository, $usersRepository, $ordersRepository);
33   
34           if ( $order == null ){
35               return new Response("Failed to find keys");
36           }
37           $em = $this->getDoctrine()->getManager();
38           $em->persist($order);
39           $em->flush();
40   
41           $balance = $usersRepository->find($_POST['pay_id'])->getBalance();
42           $res = $shopService->updateUserInfo($_POST['game_id'], $_POST['pay_id'], $balance, $_POST['amount'], $usersRepository, $keysRepository);
43           return new Response($res);

69       /**
70        * Ukáže rozdíl mezi aktuálním časem a časem, kdy uživatel vyhrál hru nebo klíč.
71        */
72       private function getTime($date)
73       {
74           $time = $date;
75           $timeNow = date_create();
76           $difference = $timeNow->diff($time);
77   
78           if($difference->d >= 1){
79               return $this->declension($difference->d, 'days');
80           }
81           else if($difference->h >= 1) {
82               return $this->declension($difference->h, 'hours');
83           }
84           else if($difference->i >= 1) {
85               return $this->declension($difference->i, 'minutes');
86           }
87           else {
88               return $this->declension($difference->s, 'seconds');
89           }
90       }

92       /**
93        * Arfografie, skloňuje čas
94        */
95       private function declension($digit, $type)
96       {
97           $str = array();
98           if($type == 'days') {
99               $str = ['день', 'дня', 'дней'];
100          }
101          else if($type == 'hours') {
102              $str = ['час', 'часа', 'часов'];
103          }
104          else if($type == 'minutes') {
105              $str = ['минуту', 'минуты', 'минут'];
106          }
107          else {
108              $str = ['секунда', 'секунды', 'секунд'];
109          }
110  
111  
112          if($digit == 1) {
113              return $digit.' '.$str[0].' назад';
114          }
115          else if($digit >= 2 && $digit <= 4) {
116              return $digit.' '.$str[1].' назад';
117          }
118          else {
119              return $digit.' '.$str[2].' назад';
120          }
121      }

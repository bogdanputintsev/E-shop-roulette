5    use App\Interfaces\ICasesRepository;

7    use Symfony\Bundle\FrameworkBundle\Controller\AbstractController;
8    use Symfony\Component\HttpFoundation\Request;
9    use Symfony\Component\HttpFoundation\Response;
10   use Symfony\Component\Routing\Annotation\Route;
11   
12   class CasesController extends AbstractController
13   {
14       /**
15        * Pro zobrazení rulety, která je navazana na nějakou hru
16        * @Route("/cases/{id}", name="cases")
17        * @param Request $request
18        * @param ICasesRepository $casesRepository
19        * @param IProductsRepository $productsRepository
20        * @return Response
21        */
22       public function index(Request $request, ICasesRepository $casesRepository, IProductsRepository $productsRepository)
23       {
24           $case = $casesRepository->find($request->get('id'));
25           $game = [];
26           $drop_sum = 0;
27   
28           if ( $case != null )
29           {
30               $game = $productsRepository->findProducts($case->getProducts());
31   
32               for($i = 0; $i < count($game); $i++){
33                   $drop_sum += $game[$i]['drop_chance'];
34               }
35           }
36   
37           return $this->render('cases/index.html.twig', [
38               'controller_name' => 'CasesController',
39               'page' => 'cases',
40               'avatar' =>   $this->get('session')->get('avatar'),
41               'uid' => 'none' ? $this->get('session')->get('id') == "" : $this->get('session')->get('id'),
42               'username' =>  $this->get('session')->get('username'),
43               'balance' => $this->get('session')->get('balance'),
44               'case_id' => $case->getId(),
45               'case_name' => $case->getName(),
46               'case_description' => $case->getDescription(),
47               'case_price' => $case->getPrice(),
48               'game' => $game,
49               'game_array_drop_sum' => $drop_sum,
50               'game_case' => 'Done' ? count($game) > 0 : 'No'
51           ]);
52       }

3    namespace App\Controller;
4    
5    use App\Interfaces\IUsersRepository;
6    use Symfony\Bundle\FrameworkBundle\Controller\AbstractController;
7    use Symfony\Component\HttpFoundation\Request;
8    use Symfony\Component\HttpFoundation\Response;
9    use Symfony\Component\Routing\Annotation\Route;
10   
11   class HandlerBonusesController extends AbstractController
12   {
13       /**
14        * Složí pro spojení s front end a obsahuje všechny metody, které jsou spojeny s bonusy.
15        * feedbackfind zajistí je-li tento uživatel napsal recenci o stránce
16        * @Route("/handler/bonuses/feedback/", name="feedbackhandler")
17        * @param IUsersRepository $usersRepository
18        * @return Response
19        */
20       public function feedbackFind(IUsersRepository $usersRepository)
21       {
22           if($usersRepository->find($_GET['uid'])->getWroteFeedback() == true){
23               return new Response("[Отзывы]: Уже использовано");
24           }
25           return new Response("[Отзывы]: Не обнаружено");
26       }
27       /**
28        * Po zahraní denní bonusové rulety, změní informaci v databází, že uživatel už nemůže víckrát hrát ruletu
29        * @Route("/handler/bonuses/stop/{id}", name="stop")
30        * @param Request $request
31        * @param IUsersRepository $usersRepository
32        * @return Response
33        */
34       public function bonusesStop(Request $request, IUsersRepository $usersRepository)
35       {
36           $usersRepository->useDailyGift($request->get('id'));
37           return new Response(1);
38       }
39   }

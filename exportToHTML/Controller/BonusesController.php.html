9    use Symfony\Bundle\FrameworkBundle\Controller\AbstractController;
10   use Symfony\Component\HttpFoundation\Response;
11   use Symfony\Component\Routing\Annotation\Route;
12   
13   class BonusesController extends AbstractController
14   {
15       /**
16        * Otevírá stránku z bonusy a sbírá všechny data pro zobrazení stránky.
17        * @Route("/bonuses", name="bonuses")
18        * @param IUsersRepository $usersRepository
19        * @param ICasesRepository $casesRepository
20        * @param IProductsRepository $productsRepository
21        * @return Response
22        */
23       public function index(IUsersRepository $usersRepository, ICasesRepository $casesRepository, IProductsRepository $productsRepository)
24       {
25           $case = $casesRepository->find(3);
26           $game = [];
27           $drop_sum = 0;
28           $gift = false;
29   
30           if ( $this->get('session')->get('id') != "" ) {
31               $gift = $usersRepository->find($this->get('session')->get('id'))->getDailyGift();
32           }
33   
34           if ( $case != null )
35           {
36               $game = $productsRepository->findProducts($case->getProducts());
37               for($i = 0; $i < count($game); $i++){
38                   $drop_sum += $game[$i]['drop_chance'];
39               }
40           }
41   
42           return $this->render('bonuses/index.html.twig', [
43               'controller_name' => 'BonusesController',
44               'page' => 'bonuses',
45               'avatar' =>   $this->get('session')->get('avatar'),
46               'uid' => 'none' ? $this->get('session')->get('id') == "" : $this->get('session')->get('id'),
47               'username' =>  $this->get('session')->get('username'),
48               'balance' => $this->get('session')->get('balance'),
49               'gift' => $gift,
50               'case_id' => 3,
51               'game' => $game,
52               'game_array_drop_sum' => $drop_sum,
53               'game_case' => 'Done' ? count($game) > 0 : 'No'
54           ]);
55       }
56   }
